



The find command is a powerful utility in Linux used 
to search for files and directories in a directory hierarchy 
based on various criteria specified by the user.


Syntax:
-------------
find [starting-point...] [expression]


[root@sanjeev ~]# mkdir makalu
[root@sanjeev ~]# cd makalu/
[root@sanjeev makalu]# touch f1.txt f2.txt f3.doc c11.xls main.c main2.cpp app1.exe
[root@sanjeev makalu]# mkdir d2
[root@sanjeev makalu]# mkdir d3
[root@sanjeev makalu]# mkdir d4
[root@sanjeev makalu]# mkdir d4/d8
[root@sanjeev makalu]# mkdir 3/d9
[root@sanjeev makalu]# mkdir d3/d9
[root@sanjeev makalu]# touch d3/d9 t1 t2 t3.txt
[root@sanjeev makalu]# mkdir d4/d8 h1.txt inc.html ki.jsp



[root@sanjeev makalu]# ls -R

-----------------------------------------------------------------
Find all files with a specific name:
-----------------------------------------------------------------


[root@sanjeev makalu]# find . -name "*.txt"
./f1.txt
./f2.txt
./t3.txt
./h1.txt



[root@sanjeev makalu]# find . -name "*.jsp"
./ki.jsp


[root@sanjeev makalu]# find . -name "*.exe"
./app1.exe
